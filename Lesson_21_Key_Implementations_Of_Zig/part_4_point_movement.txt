const std = @import("std");

const Point = struct {
    x: f32,
    y: f32,

    pub fn move(self: *Point, dx: f32, dy: f32) void {
        self.x += dx;
        self.y += dy;
    }
};

pub fn main() void {
    var p = Point{ .x = 0, .y = 0 };
    p.move(5, 10);
    std.debug.print("Point moved to: ({}, {})\n", .{p.x, p.y});
}
