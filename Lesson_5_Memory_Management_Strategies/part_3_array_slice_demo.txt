const std = @import("std");

pub fn main() !void {
    const allocator = std.heap.page_allocator;

    const size = 5;
    var arr = try allocator.alloc(u32, size);

    // Fill the array
    for (arr) |*item, index| {
        item.* = index + 1; // Fill with values 1 to 5
    }
    
    const slice: []u32 = arr[1..3]; // Taking a slice from the array
    for (slice) |item| {
        std.debug.print("{d}\n", .{item}); // Prints 2 and 3
    }

    allocator.free(arr);
}
