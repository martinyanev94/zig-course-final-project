// main.zig
const std = @import("std");
const math_operations = @import("math_operations.zig");
const string_format = @import("string_format.zig");

pub fn main() void {
    const sum = math_operations.add(5, 3);
    const product = math_operations.multiply(4, 6);
    
    std.debug.print("Sum: {}\n", .{sum});
    std.debug.print("Product: {}\n", .{product});
    
    const name = "Alice";
    const greeting = string_format.format_message(name);
    std.debug.print("{}\n", .{greeting});
}
