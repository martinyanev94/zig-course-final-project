const std = @import("std");

fn readFile(filePath: []const u8) ![]u8 {
    const file = try std.fs.cwd().openFile(filePath, .{});
    const fileLength = try file.getEndPos();
    var buffer = try std.heap.page_allocator.alloc(u8, fileLength);
    try file.readAll(buffer);
    return buffer;
}

pub fn main() !void {
    const contents = try readFile("example.txt");
    std.debug.print("File contents: {}\n", .{contents});
}
