const std = @import("std");

const Point = struct {
    x: f64,
    y: f64,
};

fn distance(a: Point, b: Point) f64 {
    const dx = a.x - b.x;
    const dy = a.y - b.y;
    return std.math.sqrt(dx * dx + dy * dy);
}

pub fn main() void {
    const p1 = Point{ .x = 1.0, .y = 1.0 };
    const p2 = Point{ .x = 4.0, .y = 5.0 };
    const d = distance(p1, p2);
    std.debug.print("Distance: {}\n", .{d});
}
