const std = @import("std");
const time = @import("std").time;

pub fn main() anyerror!void {
    var last_time = time.time_ns();
    while (true) {
        const current_time = time.time_ns();
        const delta_time = current_time - last_time;
        last_time = current_time;

        // Update game state
        update(delta_time);
        // Render game state
        render();
    }
}
